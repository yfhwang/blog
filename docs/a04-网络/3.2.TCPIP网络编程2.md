## 进程间通信

### 管道

```c
#include <unistd.h>
int pipe(int filedes[2]);
//成功返回0，失败返回-1
//filedes[0] := 管道出口
//filedes[1] := 管道入口
```

示例

```c
#include <stdio.h>
#include <unistd.h>
#define BUF_SIZE 30

int main(int argc, char *argv[])
{
	int fds[2];
	char str[]="Who are you?";
	char buf[BUF_SIZE];
	pid_t pid;
	
	pipe(fds);
	pid=fork();
	if(pid==0)
	{
		write(fds[1], str, sizeof(str));
	}
	else
	{
		read(fds[0], buf, BUF_SIZE);
		puts(buf);
	}
	return 0;
}
```
